## Usage

```tsx
import { Headline } from '@raystack/apsara/v1'

<Headline size="large" as="h1">
  Large Headline
</Headline>

<Headline size="medium">Medium Headline</Headline>

<Headline size="small" as="h3">
  Small Headline
</Headline>
```
