## Usage

```tsx
import { Grid } from "@raystack/apsara/v1";

<Grid columns={2} rows={2}>
  <Grid.Item>1</Grid.Item>
  <Grid.Item>2</Grid.Item>
  <Grid.Item>3</Grid.Item>
</Grid>;
```
